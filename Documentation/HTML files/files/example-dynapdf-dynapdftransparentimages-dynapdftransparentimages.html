Class App Inherits Application
Const kEditClear = "&Löschen"
Const kFileQuit = "Beenden"
Const kFileQuitShortcut = ""
EventHandler Sub Open() dim pdf as new MyDynapdfMBS dim f as Folderitem = SpecialFolder.Desktop.Child("DynaPDF transparent images.pdf") pdf.SetLicenseKey "Starter" // For this example you can use a Starter, Lite, Pro or Enterprise License if pdf.CreateNewPDF(f) then call pdf.SetPageCoords pdf.kpcTopDown call pdf.Append call pdf.SetCompressionFilter(pdf.kcfFlate) call pdf.SetUseTransparency true call pdf.SetSaveNewImageFormat(false) // rect background call pdf.SetColors(pdf.RGB(255,0,0)) call pdf.Rectangle(0,0,900,900,pdf.kfmFill) //draw gif file with trasparency dim gifimg as FolderItem=FindFile("img.gif") dim p as Picture=gifimg.OpenAsPicture Call pdf.InsertImageEx(10,10,0,0,gifimg,1) dim pngimg as FolderItem=Findfile("img.png") p=pngimg.OpenAsPicture Call pdf.InsertPicture(p, 10,160,0,0) call pdf.EndPage call pdf.CloseFile f.Launch true quit end if End EventHandler
Function FindFile(name as string) As FolderItem // Look for file in parent folders from executable on dim parent as FolderItem = app.ExecutableFile.Parent while parent<>Nil dim file as FolderItem = parent.Child(name) if file<>Nil and file.Exists then Return file end if parent = parent.Parent wend End Function
End Class

Case d.ActionButton Return 0 // ignore Case d.AlternateActionButton IgnoreWarnings = true Return 0 // ignore Case d.CancelButton Return -1 // stop End select End EventHandler
