EventHandler Sub Action() dim con as new SQLDatabaseMBS // connects to Microsoft SQL Server on Windows // // e.g. localhost\SQLEXPRESS@test try // connect to database // in this example it is Microsoft SQL Server, // but can also be Sybase, Informix, DB2 // Oracle, InterBase, SQLBase and ODBC dim ConnectionString as string ConnectionString = "ODBC:"+TextArea1.Text con.DatabaseName = ConnectionString con.UserName = user.text con.Password = password.text con.RaiseExceptions = true // if you like to get exceptions instead of checking error property con.Scrollable = false // disabling scrolling cursors is much faster for Microsoft SQL Server... if con.Connect then dim sql as string = "SELECT * FROM tChangedInfo;" dim r as RecordSet = con.SQLSelect(Sql) if r<>nil then MsgBox r.IdxField(1).StringValue else MsgBox "Failed to query." end if // Disconnect is optional // autodisconnect will ocur in destructor if needed con.Close end if catch r as RuntimeException MsgBox r.message end try End EventHandler
End Control
