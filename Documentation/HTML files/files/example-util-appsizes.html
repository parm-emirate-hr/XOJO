ControlInstance List Inherits Listbox
End Control
Control CalcTimer Inherits Timer
ControlInstance CalcTimer Inherits Timer
EventHandler Sub Action() // calculate size here in timer, so app is not blocked dim c as integer = List.ListCount-1 for i as integer = 0 to c if List.Cell(i,2) = "" then dim item as FolderItem = List.RowTag(i) dim d as DirectorySizeMBS = item.CalculateDirectorySizeMBS(True,10) List.Cell(i, 2) = Format(d.PhysicalTotalSize / 1000000.0, "0.0") + " MB" Return end if next End EventHandler
End Control
EventHandler Sub Open() List.ColumnAlignment(2) = List.AlignRight ListApps SpecialFolder.Applications End EventHandler
Sub ListApps(f as FolderItem) dim c as integer = f.Count for i as integer = 1 to c dim item as FolderItem = f.TrueItem(i) if item.Directory and Right(item.name,4) = ".app" then List.AddRow "", item.DisplayName List.RowPicture(List.LastIndex) = item.IconMBS(16) list.RowTag(List.LastIndex) = item end if next End Sub
End Class
