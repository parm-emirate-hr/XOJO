Function: Runs the command with sudo.
Example: 
dim s as new LinuxSuMBS

s.Command = "/usr/bin/whoami"
s.User = "root"
s.Alert = "Please Root?"
s.Message "Can I install something for you?"

  MsgBox "OK"
else
  MsgBox s.LastErrorMessage
end if
