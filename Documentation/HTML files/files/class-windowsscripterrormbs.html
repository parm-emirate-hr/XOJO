Function: The class for error details.
Example: 
dim w as new WindowsScriptMBS

w.Language = "JScript"

try
  w.AddCode "function Test(a,b) { 1 return a+b; }" // 1 is there to make error
  
  dim p(-1) as string = array("Hello ", "World from JavaScript")
  dim x as string = w.Run("Test", p)
catch e as WindowsScriptErrorExceptionMBS
  dim error as WindowsScriptErrorMBS = w.Error
  MsgBox error.Description+EndOfLine+"in "+str(error.Line)+":"+str(error.Column)
end try
