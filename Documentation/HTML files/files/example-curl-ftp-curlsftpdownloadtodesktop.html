Private Sub DoDownload(URL as string, dest as FolderItem) d = new CURLSMBS if not d.CreateMTOutputFile(dest) then MsgBox "Failed to create file." Return end if d.OptionURL = URL // optional with credentials 'd.OptionUsername = "xxx" 'd.OptionPassword = "xxx" // timeouts 'd.OptionConnectionTimeout = 10 'd.OptionTimeOut = 60 // more flags d.OptionVerbose = true d.OptionSSLVerifyHost = 0 d.OptionSSLVerifyPeer = 0 Result = d.PerformMT d.CloseMTOutputFile // show result dim s as string = d.OutputData if s.Encoding = nil then s = DefineEncoding(s, encodings.ISOLatin1) end if s = ReplaceLineEndings(s, EndOfLine) TextArea1.text = s ResultText.text = "Result: "+str(result) d = nil End Sub
