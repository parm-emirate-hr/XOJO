Sub TestCodePage(c as MemoryBlock, bit as integer, name as string) dim mask as UInt64 = Bitwise.ShiftLeft(1, bit) if BitwiseAnd(c.UInt64Value(0), mask)=mask then cList.AddRow name end if End Sub
Sub TestUnicode(c as MemoryBlock, bit as integer, name as string) if bit<64 then dim mask as UInt64 = Bitwise.ShiftLeft(1, bit) if BitwiseAnd(c.UInt64Value(0), mask)=mask then uList.AddRow name end if else bit = bit - 64 dim mask as UInt64 = Bitwise.ShiftLeft(1, bit) if BitwiseAnd(c.UInt64Value(8), mask)=mask then uList.AddRow name end if end if End Sub
