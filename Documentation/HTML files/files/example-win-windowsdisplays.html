"Height", str(display.Height) List.AddRow "Primary", yes(display.Primary) list.AddRow "" List.AddRow "DisplayAdapterActive", yes(display.DisplayAdapterActive) List.AddRow "DisplayAdapterDeviceID", display.DisplayAdapterDeviceID List.AddRow "DisplayAdapterDeviceKey", display.DisplayAdapterDeviceKey List.AddRow "DisplayAdapterDeviceName", display.DisplayAdapterDeviceName List.AddRow "DisplayAdapterDeviceString", display.DisplayAdapterDeviceString List.AddRow "DisplayAdapterRemovable", yes(display.DisplayAdapterRemovable) List.AddRow "DisplayAdapterStateFlags", str(display.DisplayAdapterStateFlags) list.AddRow "" List.AddRow "DisplayMonitorActive", yes(display.DisplayMonitorActive) List.AddRow "DisplayMonitorDeviceID", display.DisplayMonitorDeviceID List.AddRow "DisplayMonitorDeviceKey", display.DisplayMonitorDeviceKey List.AddRow "DisplayMonitorDeviceName", display.DisplayMonitorDeviceName List.AddRow "DisplayMonitorDeviceString", display.DisplayMonitorDeviceString List.AddRow "DisplayMonitorRemovable", yes(display.DisplayMonitorRemovable) List.AddRow "DisplayMonitorStateFlags", str(display.DisplayMonitorStateFlags) list.AddRow "" List.AddRow "MonitorX", str(display.MonitorX) List.AddRow "MonitorY", str(display.MonitorY) List.AddRow "MonitorWidth", str(display.MonitorWidth) List.AddRow "MonitorHeight", str(display.MonitorHeight) list.AddRow "" List.AddRow "WidthDPI", str(display.WidthDPI) List.AddRow "WidthInch", str(display.WidthInch) List.AddRow "WidthMM", str(display.WidthMM) List.AddRow "HeightDPI", str(display.HeightDPI) List.AddRow "HeightInch", str(display.HeightInch) List.AddRow "HeightMM", str(display.HeightMM) list.AddRow "" next End EventHandler
