dim c as new CGDisplayMBS
dim lines(-1) as string

dim DPIWidth  as Double = c.PixelsWide/(c.ScreenSizeWidth/10.0/2.54)

lines.Append str(c.PixelsWide)+" x "+str(c.PixelsHigh)+" Pixel is"
