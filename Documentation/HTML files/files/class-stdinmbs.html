Function: A class to read from stdin.
Example: 
dim s as new StdoutMBS
dim r as new StdinMBS
dim v,n as Integer

s.Write "Integer: "
n=r.ReadInteger(v)

print "integer read: "+str(v)
print "integer count: "+str(n)

dim d as Double

s.Write "Double: "
n=r.ReadDouble(d)

print "double read: "+str(d)
print "double count: "+str(n)

dim t as string

s.Write "String: "
n=r.ReadString(t)

print "string read: "+t
print "string count: "+str(n)

do
  s.Write "Character (type a and return to end): "
  n=r.GetCharacter
  
  print "got character: "+str(n)
loop until n=65 or n=97
