EventHandler Sub Action() // this is for Mac where we detect app by Bundle ID const BundleID = "com.apple.Safari" // step 1, quit Safari dim found as Boolean = false dim p as new ProcessMBS p.GetFirstProcess do if p.BundleID = BundleID then found = true call p.QuitProcess exit end if loop until not p.GetNextProcess if found then // step 2, wait for Safari to quit do found = false p.GetFirstProcess do if p.BundleID = BundleID then found = true end if loop until not p.GetNextProcess if not found then // wait some time DelayMBS 0.1 end if loop until found = false end if // step 3, start again // requires also Mac plugin dim file as FolderItem = LaunchServicesFindApplicationForInfoMBS("", BundleID, "") // LaunchServicesFindApplicationForInfoMBS is in MacOSX plugin if file<>Nil then file.Launch end if End EventHandler
End Control
End Class
