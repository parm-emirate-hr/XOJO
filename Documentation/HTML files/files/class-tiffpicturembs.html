Function: A class for a Tiff picture.
Example: 
Dim p As Picture = LogoMBS(500)

// save tiff file

Dim t As New TiffPictureMBS
Dim file4 As FolderItem = SpecialFolder.Desktop.Child("test4.tif")

t.Pict = p

If t.Create(file4) Then
  If t.WriteRGB Then
    t.Close
    // ok
  Else
    Break
  End If
Else
  Break
End If
