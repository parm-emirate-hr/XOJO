Sub update() p=new PictureMBS(Width,Height,ImageFormat) if mode=0 then call p.ScaleFast(logo, Width, Height) else // we need a temporary picture // provide the temp picture outself: 'dim temp as new PictureMBS(Width, p.Height, PictureMBS.ImageFormatScaling) ' 'if temp.Valid=false then 'MsgBox "Out of memory" 'quit 'else 'call p.Scale(logo, temp, mode, Width, Height) 'end if // or do it automatically using the PictureFactoryMBS: if p.Scale(logo, nil, mode, Width, Height)=false then MsgBox "Out of memory" quit end if end if Backdrop=p.CopyPicture Title=str(Width)+" x "+str(Height)+" with mode "+str(mode) End Sub
