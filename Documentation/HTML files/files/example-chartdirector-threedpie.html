EventHandler Sub Open() // The data for the pie chart dim data(-1) as double = array(25, 18, 15, 12, 8, 30, 35.0) // The labels for the pie chart dim labels(-1) as string = Array("Labor", "Licenses", "Taxes", "Legal", "Insurance", "Facilities", "Production") // Create a PieChart object of size 360 x 300 pixels dim c as new CDPieChartMBS(360,300) // Set the center of the pie at (180, 140) and the radius to 100 pixels c.setPieSize(180, 140, 100) // Add a title to the pie chart call c.addTitle("Project Cost Breakdown") // Draw the pie in 3D c.set3D // Set the pie data and the pie labels c.setData(data,labels) // Explode the 1st sector (index = 0) c.setExplode(0) Backdrop=c.makeChartPicture End EventHandler
End Class
