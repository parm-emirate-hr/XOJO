EventHandler Sub Action() dim l as integer if searching then if HTMLViewer1.IEContinueFindTextMBS(EditField1.Text, 0, 0, true) then MsgBox "Found" else searching=False MsgBox "not found" end if else if HTMLViewer1.IEFindTextMBS(EditField1.Text, 0, 0, true) then MsgBox "Found" searching=True else MsgBox "not found" end if end if End EventHandler
End Control

End Control
