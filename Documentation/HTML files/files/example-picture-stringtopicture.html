Function StringToPicture(data as string) As Picture // self made String To Picture method without QuickTime // needs TIFF, JPEG, Picture, PNG and GIF Plugins if lenb(data)<10 then Return nil dim m as MemoryBlock = leftb(data, 10) Select case m.UInt8Value(0) case &h42 // BMP // from Picture plugin Return BMPStringtoPictureMBS(data) case &h47 // GIF // from GIF plugin Return GifStringToPictureMBS(data) case &h49 // I, Tiff for Intel format // from tiff plugin Return TIFFStringToPictureMBS(data) case &h4D // M, Tiff for Motorola (PPC) format Return TIFFStringToPictureMBS(data) case &hFF // JPEG // from JPEG plugins Return JPEGStringToPictureMBS(data, true) end Select End Function
End Class
