Function: A class for detailed error information.
Example: 
// a simply example using the AppleScriptErrorMBS class:

dim s As new AppleScriptMBS
dim ASErr As AppleScriptErrorMBS

dim text As string = "seet x to 1" // intentional error

s.Compile text

if s.LastError<>0 Then
  dim errorMsg As String = "Compile error: " + str(s.Lasterror) + EndOfLine
  ASErr = s.Error
  MsgBox errorMsg + "Details: "  + ASErr.BriefMessage
end if
