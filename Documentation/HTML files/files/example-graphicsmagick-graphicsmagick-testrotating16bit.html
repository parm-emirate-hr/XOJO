'image.filterType = image.LanczosFilter 'image.scale(new GMGeometryMBS(100,80)) dim co as new GM16ColorRGBMBS(1.0, 0.0, 0.0) image.BackgroundColor=co image.Rotate(50.0) // save 16 bit dim blob as new GM16BlobMBS image.depth = 16 image.write blob f = SpecialFolder.Desktop.Child("output16.tiff") dim b as BinaryStream b = f.CreateBinaryFile("application/binary") b.Write blob.CopyString b.Close // save 8 bit blob = new GM16BlobMBS image.depth = 8 image.write blob f = SpecialFolder.Desktop.Child("output8.tiff") b = f.CreateBinaryFile("application/binary") b.Write blob.CopyString b.Close // show Backdrop=image.CopyPicture(0,0,image.columns, image.rows) MsgBox "Saved "+f.Name+" on your desktop." End EventHandler
