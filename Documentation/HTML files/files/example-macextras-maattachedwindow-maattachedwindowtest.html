EventHandler Sub Action() dim w as NSWindowMBS = self.NSWindowMBS if m<>Nil then w.removeChildWindow m m.hide m = nil end if dim v as NSViewMBS = me.NSViewMBS dim side as integer = MAAttachedWindowMBS.MAPositionAutomatic dim distance as Double = 10.0 dim p as NSPointMBS = new NSPointMBS(me.left + me.Width/2, me.top + me.Height/2) dim tv as new NSTextFieldMBS(10,10, 180, 50) tv.StringValue = "Hello World" tv.Editable = false tv.textColor = NSColorMBS.whiteColor tv.backgroundColor = NSColorMBS.blackColor tv.Bordered = false dim vv as new NSViewMBS(0,0,200,70) vv.addSubview tv m = MAAttachedWindowMBS.attachedWindow(vv, p, w, side, distance) m.borderColor = NSColorMBS.whiteColor m.backgroundColor = NSColorMBS.blackColor m.viewMargin = 4 m.borderWidth = 3 m.cornerRadius = 8 m.drawsRoundCornerBesideArrow = true m.arrowBaseWidth = 20 m.arrowHeight = 15 m.hasArrow = true m.alphaValue = 0.9 w.addChildWindow m, NSWindowMBS.NSWindowAbove m.show End EventHandler
End Control
