EventHandler Sub Open() // use internal sqlite library call InternalSQLiteLibraryMBS.Use dim db as new SQLDatabaseMBS // where is the library? 'db.Option(SQLConnectionMBS.kOptionLibrarySQLite) = "/usr/lib/libsqlite3.0.dylib" // connect to database // in this example it is SQLite, // but can also be Sybase, Oracle, Informix, DB2, SQLServer, InterBase, MySQL, SQLBase and ODBC dim path as string if TargetMacOS then #if RBVersion < 2013 then path = GetFolderItem("test.db").UnixpathMBS // put the database in the current folder #else path = GetFolderItem("test.db").NativePath // put the database in the current folder #endif else path = "test.db" // for Windows and Linux in the current folder the application is inside. end if // set the key db.SQLiteEncryptionKey = "hello" // set the path db.DatabaseName = "sqlite:"+path // and connect if db.Connect then MsgBox "We are connected!" // create table db.SQLExecute "Create table test_tbl(fid integer, fvarchar20 varchar(20), fblob blob)" // insert value db.SQLExecute "Insert into test_tbl(fid, fvarchar20) values (1, 'Some string (1)')" MsgBox("Table created, row inserted!") end if End EventHandler
End Class
