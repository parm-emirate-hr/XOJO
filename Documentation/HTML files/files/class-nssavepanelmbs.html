Function: The class for a Cocoa Save Panel.
Example: 

dim n as new NSSavePanelMBS

dim e as integer = n.runModal

if e = n.NSCancelButton then
  // cancelled
  Break
elseif e = n.NSOKButton then
  // got file
  dim file as FolderItem = n.file
  Break
end if
